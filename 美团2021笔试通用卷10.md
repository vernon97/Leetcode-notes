```cpp
#include <iostream>
#include <algorithm>
#include <cstring>
#include <vector>
using namespace std;

int main()
{
    int n, m, a, b;
    while(cin >> n >> m >> a >> b)
    {
        vector<int> nums;
        int x, minv = 0x3f3f3f, maxv= -0x3f3f3f3f;
        int equala = 0, equalb = 0;
        bool success = true;
        if(a > b) swap(a, b);
        for(int i = 0; i < m ; i++)
        {
            cin >> x;
            nums.push_back(x);
            minv = min(minv, x);
            maxv = max(maxv, x);
            if(x == a) equala = 1;
            if((a != b || equala)&& x == b) equalb = 1;
        }
        // 不行的几个情况 
        if(minv < a || maxv > b) success = false;
        int left = 2 - equala - equalb;
        if(left > n - m) success = false;
        if(success) puts("YES");
        else puts("NO");
    }
    return 0;
}
```

```cpp
#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;
int n, m;
int main()
{
    cin >> n >> m;
    int x, cnt = 0;
    vector<int> f;
    for(int i = 0; i < n; i++)
    {
        cin >> x;
        f.push_back(x);
        if(x > 0) cnt++;
    }
    sort(f.begin(), f.end());
    int threshold = f[n - m];
    if(threshold == 0) cout << cnt << endl;
    else{
        auto it = find(f.begin(), f.end(), threshold);
        cout <<distance(it, f.end()) << endl;
    }
    return 0;
}
```

```cpp
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cstdio>
using namespace std;

const int N = 100010;

int n;
int s[2 * N];
int q[N];

int main()
{
    int T;
    scanf("%d", &T);
    while(T--)
    {
        int hh = 0, tt = 0; // 默认有一个s[0]
        scanf("%d", &n);
        for(int i = 1; i <= n; i++)
        {
            scanf("%d", s + i);
            s[i + n] = s[i];
        }
        q[0] = 0;
        // 构建前缀和
        for(int i = 1; i <= 2 * n; i++)
            s[i] += s[i - 1];
        int res = 0;
        for(int i = 1; i <= 2 * n; i++)
        {
            if(hh <= tt && q[hh] < i - n) hh++;
            while(hh <= tt && s[q[tt]] >= s[i]) tt--;
            res = max(res, s[i] - s[q[hh]]);
            q[++tt] = i;
        }
        cout << res << endl;
    }
    return 0;
}
```

```cpp
#include <iostream>
#include <cstdio>
#include <algorithm>
#include <unordered_map>
#include <cstring>
using namespace std;

const int N = 5050, M = 2 * N;

int n, cnt, m;
int h[N], e[M], w[M], ne[M], idx;
int ans[N];
unordered_map<int, int> ha, reha;
int f[N][N];
void add(int a, int b)
{
    e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

int dfs(int root, int color, int father)
{
    if(f[root][color] != -1) return f[root][color];
    // 往下搜 统计每个颜色的出现次数
    int curcnt = (w[root] == color);
    for(int i = h[root]; ~i; i = ne[i])
    {
        int j = e[i];
        if(j == father) continue;
        curcnt += dfs(j, color, root);
    }
    f[root][color] = curcnt;
    return curcnt;
}

int main()
{
    memset(h, -1, sizeof h);
    memset(f, -1, sizeof f);
    scanf("%d", &n);
    for(int i = 1; i < n; i++)
    {
        int a, b;
        scanf("%d%d", &a, &b);
        add(a, b), add(b, a);
    }
    for(int i = 1; i <= n; i++)
    {
        int c;
        scanf("%d", &c);
        if(!ha.count(c))
        {
            ha[c] = cnt++;
            reha[ha[c]] = c;
        }
        w[i] = ha[c]; // 离散化
    }

    // 显然是dfs了 -> 记忆化搜索
    scanf("%d", &m);
    for(int c = 0; c < cnt; c++)
        dfs(1, c, -12);
    while(m--)
    {
        int q;
        scanf("%d", &q);
        int maxv = 0, ans = 0;
        for(int c = 0; c < cnt; c++)
        {
            int curcnt = f[q][c];
            if(curcnt > maxv)
            {
                ans = reha[c];
                maxv = curcnt;
            }
        }
        cout << ans << endl;
    }
    //cout << "--------" << endl;

     //   for(int j = 0; j < cnt; j++)
      //      cout << f[i][j] << ' ';

    return 0;
}
```
